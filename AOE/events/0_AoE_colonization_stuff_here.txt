country_event = {
	id = 19180007
	title = "$FROMCOUNTRY$ 's punitive expedition"
	desc = "EVENTDESC19180007"
	picture = "colonization"
	is_triggered_only = yes
	
	option = {
		name = "They will regret coming here"
		add_country_modifier = {
			name = fanatic_army
			duration = 1825
		}
		FROM = {
			country_event = 19180008
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				military_score = 0
			}
		}
	}
	
	option = {
		name = "We will give up our lands"
		FROM = {
			any_owned = {
				random_neighbor_province = {
					limit = {
						owned_by = THIS
						is_core = FROM
					}
					secede_province = FROM
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 15
				military_score = 0
			}
		}
	}
}
